# 현재시간 표기(년, 월, 일, 시) ex)2023090112 / API에서 date값 주고받는 양식
def current_TimeNow():
    current_time = datetime.now() 
    formatted_time = current_time.strftime("%Y%m%d%H")
    return formatted_time

# 현재시간 ~ 00시까지 계산 ex) 12시~00시까지 개수 12개
def disPlayTimeCount(datet):
    splitdatet = datet[-2::] # ex)2023090109에서 9,10번째 값(시간)
    timeCount = 24 - int(splitdatet) + 1 # 24는 00시 기준, 현재시간 포함 +1
    return timeCount
